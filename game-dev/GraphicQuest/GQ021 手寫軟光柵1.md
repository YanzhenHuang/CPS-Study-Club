# 手寫軟光柵1
就是使用C++寫一個小小的渲染器，不使用圖形庫

# 第一步

```c++
#include <iostream>

int main()
{
	const int image_width = 256;
	const int image_height = 256;

	std::cout << "P3\n" << image_width << " " << image_height << "\n255\n";
	
	for (int i = image_height - 1; i >= 0; --i)
	{
		for (int j = 0; j < image_width; ++j)
		{
			auto r = double(j) / (image_width - 1);
			auto g = double(i) / (image_height - 1);
			auto b = 0.25;

			int ir = static_cast<int>(255.999 * r);
			int ig = static_cast<int>(255.999 * g);
			int ib = static_cast<int>(255.999 * b);

			std::cout << ir << ' ' << ig << ' ' << ib << '\n';
		}
	}
}
```

# 第二步

定義三維向量類 vec3.h

```c++
#ifndef VEC3_H
#define VEC3_H

#include <cmath>
#include <iostream>

using std::sqrt;

class vec3 {
public:
	vec3() : e{0,0,0} {}
	vec3(double e0, double e1, double e2) : e{ e0,e1,e2 } {};
	
	double x() const { return e[0]; }
	double y() const { return e[1]; }
	double z() const { return e[2]; }

	// 負向量
	vec3 operator-() const { return vec3(-e[0], -e[1], e[2]); }
	
	double operator[](int i) const { return e[i]; }
	double& operator[](int i) { return e[i]; }

	// 向量加法
	vec3& operator +=(const vec3 &v)
	{
		e[0] += v.e[0];
		e[1] += v.e[1];
		e[2] += v.e[2];
		return *this;
	}

	// 向量乘法
	vec3& operator *=(const double t)
	{
		e[0] *= t;
		e[1] *= t;
		e[2] *= t;
		return *this;
	}

	// 向量除法
	vec3& operator /=(const double t)
	{
		return *this *= 1 / t;
	}

	// 向量長度
	double length() const {
		return sqrt(length_squared());
	}

	// 向量xyz平方和
	double length_squared() const {
		return e[0] * e[0] + e[1] * e[1] + e[2] * e[2];
	}

public:
	double e[3];
};

using point3 = vec3; //	3D 點
using color = vec3; // RGB 顏色

// 通用函數列表

// 輸出向量e1e2e3
inline std::ostream& operator<<(std::ostream& out, const vec3& v) {
	return out << v.e[0] << ' ' << v.e[1] << ' ' << v.e[2];
}

// 向量與向量的+-*

inline vec3 operator+(const vec3& u, const vec3& v) {
	return vec3(u.e[0] + v.e[0], u.e[1] + v.e[1], u.e[2] + v.e[2]);
}

inline vec3 operator-(const vec3& u, const vec3& v) {
	return vec3(u.e[0] - v.e[0], u.e[1] - v.e[1], u.e[2] - v.e[2]);
}

inline vec3 operator*(const vec3& u, const vec3& v) {
	return vec3(u.e[0] * v.e[0], u.e[1] * v.e[1], u.e[2] * v.e[2]);
}

// 乘以常數

inline vec3 operator*(double t, const vec3& v)
{
	return vec3(t * v.e[0], t * v.e[1], t * v.e[2]);
}

inline vec3 operator*(const vec3& v, double t)
{
	return t*v;
}

inline vec3 operator/(vec3& v, double t)
{
	return (1/t) * v;
}

// 點乘
inline double dot(const vec3& u, const vec3& v)
{
	return u.e[0] * v.e[0] + u.e[1] * v.e[1] + u.e[2] * v.e[2];
}

// 外積 
// https://zh.wikipedia.org/zh-tw/%E5%8F%89%E7%A7%AF
inline vec3 cross(const vec3& u, const vec3& v)
{
	return vec3(u.e[1] * v.e[2] - u.e[2] * v.e[1], u.e[2] * v.e[0] - u.e[0] * v.e[2], u.e[0] * v.e[1] - u.e[1] * v.e[0]);
}

inline vec3 unit_vector(vec3 v) {
	return v / v.length();
}

#endif VEC3_H
```

顏色類 color.h

```c++
#ifndef COLOR_H
#define COLOR_H

#include "vec3.h"

#include <iostream>

void write_color(std::ostream& out, color pixel_color) {
    // Write the translated [0,255] value of each color component.
    out << static_cast<int>(255.999 * pixel_color.x()) << ' '
        << static_cast<int>(255.999 * pixel_color.y()) << ' '
        << static_cast<int>(255.999 * pixel_color.z()) << '\n';
}

#endif
```

改寫我們主要函數:

```c++
#include "color.h"
#include "vec3.h"
#include <iostream>

int main()
{
	const int image_width = 256;
	const int image_height = 256;

	std::cout << "P3\n" << image_width << " " << image_height << "\n255\n";

	for (int i = image_height - 1; i >= 0; --i)
	{
        // 進度條，輸出在 cerr 錯誤輸出台上，不影響cout
		std::cerr << "\rScanlines remaining: " << i << ' ' << std::flush;

		for (int j = 0; j < image_width; ++j)
		{
			color pixel_color(double(j) / (image_width - 1),
				double(i) / (image_height - 1),
				0.25);
			write_color(std::cout, pixel_color);
		}
	}
	std::cerr << "\nDone.\n";
}
```
